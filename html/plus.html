Popup弹窗

JS提示框说明：
JS调用：
$.popup({
	title:'系统提示',
    str: '系统提示信息！', //弹窗文字
    icon : '', //默认标记 有4个值：success，warning，danger，loading
    time : null, //自动关闭,  如果有值，一定时间会自动关闭
    clickback : false ,//点击按钮的回调函数 return :mark,$da
    closeback : false, //关闭时的回调函数  return :$da
    bgshow : true, //是否要显示半透明的黑色背景？
    color: '#000',// 背景黑色 和白色
    fadetime : 500, //渐变动画时间
    showbtn : false, //是否显示按钮 false,1,2
    width: 200, //弹窗宽度
    height: 'auto', //弹窗高度
    drag : false, //是否可拖动
    close: true //是否要点击背景关闭？
});

分页

//分页插件调用
//分页
var nowp = $("#pageList1").pagelist({
      callback: function(page,maxcount){
      	console.log("您选择了页码：",page);
      	ajaxDo(page);
      }
});
//ajax方法
var ajaxDo = function(page,maxcount){
	console.log("ajax doing..."+page);

	//模拟ajax方法
	setTimeout(function(){
    	var data = {
    		nowpage:page,
		    count:300,
		    maxcount:maxcount
    	}
    	nowp.refresh(data);
    },500);
}
//默认加载第一页
ajaxDo(1,20);
